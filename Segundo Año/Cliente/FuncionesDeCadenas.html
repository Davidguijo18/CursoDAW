1. Convertir a mayúsculas function toUpperCase(str) { return str.toUpperCase(); } // 2. Convertir a minúsculas function toLowerCase(str) { return str.toLowerCase(); } // 3. Capitalizar la primera letra de una cadena function capitalizeFirstLetter(str) { return str.charAt(0).toUpperCase() + str.slice(1); } // 4. Capitalizar cada palabra en una cadena function capitalizeWords(str) { return str.split(' ').map(word => capitalizeFirstLetter(word)).join(' '); } // 5. Eliminar espacios en blanco al inicio y al final function trimWhitespace(str) { return str.trim(); } // 6. Eliminar todos los espacios en blanco function removeSpaces(str) { return str.replace(/\s+/g, ''); } // 7. Verificar si una cadena contiene otra cadena function containsSubstring(str, substring) { return str.includes(substring); } // 8. Obtener la longitud de una cadena function getLength(str) { return str.length; } // 9. Reemplazar una subcadena con otra function replaceSubstring(str, oldSubstring, newSubstring) { return str.replace(new RegExp(oldSubstring, 'g'), newSubstring); } // 10. Contar la cantidad de ocurrencias de un substring function countOccurrences(str, substring) { const regex = new RegExp(substring, 'g'); const matches = str.match(regex); return matches ? matches.length : 0; } // 11. Invertir una cadena function reverseString(str) { return str.split('').reverse().join(''); } // 12. Verificar si una cadena es un palíndromo function isPalindrome(str) { const normalizedStr = str.toLowerCase().replace(/[^a-z0-9]/g, ''); return normalizedStr === reverseString(normalizedStr); } // 13. Extraer una subcadena function substring(str, start, end) { return str.substring(start, end); } // 14. Obtener el índice de la primera aparición de un carácter function indexOfChar(str, char) { return str.indexOf(char); } // 15. Dividir una cadena en un array de subcadenas function splitString(str, delimiter) { return str.split(delimiter); } // 16. Unir un array de cadenas en una sola cadena function joinStrings(arr, delimiter) { return arr.join(delimiter); } // 17. Repetir una cadena varias veces function repeatString(str, count) { return str.repeat(count); } // 18.
